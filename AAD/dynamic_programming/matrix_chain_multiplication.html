Matrix Chain Multiplication

Problem Statement:

Algorithm

Code : 

    // matrix chain multiplication

    #include <bits/stdc++.h>
    
    using namespace std;
    
    int mcm(int order[], int i, int j)
    {
    
        if (i == j)
        {
            return 0;
        }
        int min = INT_MAX;
        for (int k = i; k < j; k++)
        {
            int count = mcm(order, i, k) + mcm(order, k + 1, j) + order[i - 1] * order[k] * order[j];
            if (count < min)
            {
                min = count;
            }
        }
        return min;
    }
    
    int main()
    {
        int n;
        cin >> n;
        int order[n];
        for (int i = 0; i < n; i++)
        {
            cin >> order[i];
        }
        cout<<" min multiplications = "<<mcm(order,1,n-1)<<"\n";
    }

Complexity Analysis :

Time complexity : \(O(n^3)\)
